function uid(prefix = "id") {
  return `${prefix}_${Date.now()}_${Math.random().toString(16).slice(2)}`;
}

function decodeHTMLEntities(raw: string): string {
  if (typeof window === "undefined") return raw;
  const ta = document.createElement("textarea");
  ta.innerHTML = raw;
  return ta.value;
}

function countMatches(input: string, regex: RegExp): number {
  const m = input.match(regex);
  return m ? m.length : 0;
}

function detectEscapedHTML(input: string): boolean {
  const escapedCount = countMatches(input, /&lt;|&gt;/gi);
  const angleCount = countMatches(input, /[<>]/g);
  return escapedCount >= 4 || (escapedCount >= 2 && escapedCount > angleCount);
}

function looksLikeHTML(s: string): boolean {
  return /<[a-z][\s\S]*?>/i.test(s);
}

function openDB(): Promise<IDBDatabase> {
  return new Promise((resolve, reject) => {
    if (typeof indexedDB === "undefined") {
      reject(new Error("indexedDB unavailable"));
      return;
    }
    const req = indexedDB.open(NB_DB, 1);
    req.onupgradeneeded = () => {
      const db = req.result;

      if (!db.objectStoreNames.contains(STORE_INK)) {
        db.createObjectStore(STORE_INK, { keyPath: "id" }); // id = notebookId
      }
      if (!db.objectStoreNames.contains(STORE_DOC)) {
        db.createObjectStore(STORE_DOC, { keyPath: "id" }); // id = notebookId
      }
      if (!db.objectStoreNames.contains(STORE_ANN)) {
        const store = db.createObjectStore(STORE_ANN, { keyPath: "id" }); // id = annotationId
        store.createIndex("byNotebook", "notebookId", { unique: false });
      }
    };
    req.onsuccess = () => resolve(req.result);
    req.onerror = () => reject(req.error ?? new Error("IndexedDB open failed"));
  });
}

async function dbGet<T>(storeName: string, key: any): Promise<T | undefined> {
  const db = await openDB();
  const out = await new Promise<T | undefined>((resolve, reject) => {
    const tx = db.transaction(storeName, "readonly");
    const store = tx.objectStore(storeName);
    const req = store.get(key);
    req.onsuccess = () => resolve(req.result as T | undefined);
    req.onerror = () => reject(req.error ?? new Error("IDB read failed"));
  });
  db.close();
  return out;
}

async function dbPut(storeName: string, value: any): Promise<void> {
  const db = await openDB();
  await new Promise<void>((resolve, reject) => {
    const tx = db.transaction(storeName, "readwrite");
    tx.objectStore(storeName).put(value);
    tx.oncomplete = () => resolve();
    tx.onerror = () => reject(tx.error ?? new Error("IDB write failed"));
  });
  db.close();
}

async function dbDelete(storeName: string, key: any): Promise<void> {
  const db = await openDB();
  await new Promise<void>((resolve, reject) => {
    const tx = db.transaction(storeName, "readwrite");
    tx.objectStore(storeName).delete(key);
    tx.oncomplete = () => resolve();
    tx.onerror = () => reject(tx.error ?? new Error("IDB delete failed"));
  });
  db.close();
}

async function dbQueryAnnotations(notebookId: number): Promise<Annotation[]> {
  const db = await openDB();
  const out = await new Promise<Annotation[]>((resolve, reject) => {
    const tx = db.transaction(STORE_ANN, "readonly");
    const store = tx.objectStore(STORE_ANN);
    const idx = store.index("byNotebook");
    const req = idx.getAll(notebookId);
    req.onsuccess = () => resolve((req.result as Annotation[]) ?? []);
    req.onerror = () => reject(req.error ?? new Error("IDB query failed"));
  });
  db.close();
  return out;
}

function drawStroke(ctx: CanvasRenderingContext2D, stroke: InkStroke): void {
  if (stroke.points.length < 2) return;
  ctx.save();
  ctx.lineCap = "round";
  ctx.lineJoin = "round";

  if (stroke.tool === "eraser") {
    ctx.globalCompositeOperation = "destination-out";
    ctx.globalAlpha = 1;
    ctx.strokeStyle = "rgba(0,0,0,1)";
    ctx.lineWidth = stroke.size;
  } else if (stroke.tool === "highlighter") {
    ctx.globalCompositeOperation = "source-over";
    ctx.globalAlpha = 0.28;
    ctx.strokeStyle = stroke.color;
    ctx.lineWidth = stroke.size * 2.0;
  } else {
    ctx.globalCompositeOperation = "source-over";
    ctx.globalAlpha = 1;
    ctx.strokeStyle = stroke.color;
    ctx.lineWidth = stroke.size;
  }

  ctx.beginPath();
  ctx.moveTo(stroke.points[0].x, stroke.points[0].y);
  for (let i = 1; i < stroke.points.length; i += 1) ctx.lineTo(stroke.points[i].x, stroke.points[i].y);
  ctx.stroke();
  ctx.restore();
}

function ensureBridgeStyle(doc: Document) {
  const head = doc.head || doc.getElementsByTagName("head")[0];
  if (!head) return;

  // If notebook already has bridge style, skip.
  if (head.querySelector('style[data-nb-bridge="1"]')) return;

  const style = doc.createElement("style");
  style.setAttribute("data-nb-bridge", "1");
  style.textContent = IFRAME_BRIDGE_CSS;
  head.appendChild(style);
}

function collectTOC(doc: Document): TocEntry[] {
  const toc: TocEntry[] = [];
  let c = 0;
  doc.body.querySelectorAll("h1,h2,h3").forEach((el) => {
    const text = (el.textContent ?? "").trim();
    if (!text) return;
    c += 1;
    if (!el.id) el.id = `nb-heading-${c}`;
    toc.push({ id: el.id, text, level: Number(el.tagName[1] || 1) });
  });
  return toc;
}

function unwrap(el: Element) {
  const parent = el.parentNode;
  if (!parent) return;
  while (el.firstChild) parent.insertBefore(el.firstChild, el);
  parent.removeChild(el);
}

function serializeDoc(doc: Document): string {
  // Clone the DOM, remove ephemeral search markers, keep user highlights/pins.
  const clone = doc.documentElement.cloneNode(true) as HTMLElement;
  clone.querySelectorAll(".nb-search-hit").forEach((hit) => unwrap(hit));
  // Keep bridge CSS; itâ€™s harmless and ensures pins/highlights keep styling.
  return "<!doctype html>\n" + clone.outerHTML;
}

function clearSearchHits(doc: Document) {
  doc.querySelectorAll(".nb-search-hit").forEach((hit) => unwrap(hit));
}

function applySearch(doc: Document, query: string): { count: number; nodes: Element[] } {
  clearSearchHits(doc);
  const q = query.trim();
  if (q.length < 2) return { count: 0, nodes: [] };

  const escaped = q.replace(/[.*+?^${}()|[\]\\]/g, "\\$&");
  const re = new RegExp(escaped, "gi");

  const walker = doc.createTreeWalker(doc.body, NodeFilter.SHOW_TEXT, {
    acceptNode(node) {
      const t = node.nodeValue ?? "";
      if (!t || t.trim().length === 0) return NodeFilter.FILTER_REJECT;
      if (!re.test(t)) return NodeFilter.FILTER_REJECT;
      // reset lastIndex after test
      re.lastIndex = 0;
      return NodeFilter.FILTER_ACCEPT;
    },
  });

  const hits: Element[] = [];
  const textNodes: Text[] = [];
  while (walker.nextNode()) textNodes.push(walker.currentNode as Text);

  for (const tn of textNodes) {
    const text = tn.nodeValue ?? "";
    if (!text) continue;

    const parent = tn.parentElement;
    if (!parent) continue;
    // Skip inside script/style just in case
    const tag = parent.tagName?.toLowerCase();
    if (tag === "script" || tag === "style") continue;

    const frag = doc.createDocumentFragment();
    let last = 0;
    let m: RegExpExecArray | null;

    re.lastIndex = 0;
    while ((m = re.exec(text)) !== null) {
      const start = m.index;
      const end = start + m[0].length;

      if (start > last) frag.appendChild(doc.createTextNode(text.slice(last, start)));

      const span = doc.createElement("span");
      span.className = "nb-search-hit";
      span.textContent = text.slice(start, end);
      frag.appendChild(span);
      hits.push(span);

      last = end;
    }
    if (last < text.length) frag.appendChild(doc.createTextNode(text.slice(last)));

    parent.replaceChild(frag, tn);
  }

  return { count: hits.length, nodes: hits };
}

  const selectionRangeRef = useRef<Range | null>(null);

  // Paper refs
  const paperViewportRef = useRef<HTMLDivElement>(null);
  const paperInnerRef = useRef<HTMLDivElement>(null);

  // Iframe + canvas overlay
  const iframeRef = useRef<HTMLIFrameElement>(null);
  const canvasRef = useRef<HTMLCanvasElement>(null);

  const isDrawingRef = useRef(false);
  const currentStrokeRef = useRef<InkStroke | null>(null);

  const rawContent = String(notebook?.content ?? "");

  // Build initial srcDoc: prefer stored doc override, else use notebook.content.
  const [srcDoc, setSrcDoc] = useState<string>("");

  const baseDoc = useMemo(() => {
    // Create a robust HTML document string that preserves CSS of imported HTML.
    if (!rawContent) return "";

    const escaped = detectEscapedHTML(rawContent);
    const decoded = escaped ? decodeHTMLEntities(rawContent) : rawContent;

    // If it's already a full doc, keep it
    const hasHtmlTag = /<html[\s>]/i.test(decoded) || /<!doctype/i.test(decoded);
    if (hasHtmlTag) return decoded;

    // If it looks like an HTML fragment, wrap it
    if (looksLikeHTML(decoded)) return makeDocFromFragment(decoded, notebook?.title || "Notebook");

    // Otherwise treat as plain text and wrap in paragraphs
    const safeText = decoded
      .replace(/&/g, "&amp;")
      .replace(/</g, "&lt;")
      .replace(/>/g, "&gt;");
    const html = safeText
      .split(/\n{2,}/)
      .map((block) => `<p>${block.replace(/\n/g, "<br/>")}</p>`)
      .join("");

    return makeDocFromFragment(html, notebook?.title || "Notebook");
  }, [rawContent, notebook?.title]);

  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  // Load notebook
  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  useEffect(() => {
    if (!Number.isFinite(nbId) || nbId <= 0) {
      setNotebook(null);
      return;
    }
    let cancelled = false;
    getNotebook(nbId).then((nb) => {
      if (!cancelled) setNotebook(nb ?? null);
    });
    return () => {
      cancelled = true;
    };
  }, [nbId]);

  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  // Fullscreen state sync
  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  useEffect(() => {
    const onFs = () => setIsFullscreen(Boolean(document.fullscreenElement));
    document.addEventListener("fullscreenchange", onFs);
    return () => document.removeEventListener("fullscreenchange", onFs);
  }, []);

  const toggleFullscreen = useCallback(async () => {
    const el = fullscreenRootRef.current;
    if (!el) return;
    try {
      if (!document.fullscreenElement) await el.requestFullscreen();
      else await document.exitFullscreen();
    } catch {
      // ignore
    }
  }, []);

  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  // Load persisted: doc html + ink + annotations
  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  useEffect(() => {
    if (!Number.isFinite(nbId) || nbId <= 0) return;

    let cancelled = false;

    (async () => {
      // DOC
      try {
        const docRow = await dbGet<{ id: number; html: string }>(STORE_DOC, nbId);
        const html = docRow?.html || localStorage.getItem(LS_DOC(nbId)) || "";
        if (!cancelled) setSrcDoc(html || baseDoc);
      } catch {
        const html = localStorage.getItem(LS_DOC(nbId)) || "";
        if (!cancelled) setSrcDoc(html || baseDoc);
      }

      // INK
      try {
        const inkRow = await dbGet<{ id: number; strokes?: InkStroke[] }>(STORE_INK, nbId);
        const strokes = Array.isArray(inkRow?.strokes) ? inkRow!.strokes! : [];
        if (!cancelled) {
          setInkStrokes(strokes);
          setRedoStack([]);
        }
      } catch {
        try {
          const raw = localStorage.getItem(LS_INK(nbId));
          const parsed = raw ? (JSON.parse(raw) as InkStroke[]) : [];
          if (!cancelled) {
            setInkStrokes(Array.isArray(parsed) ? parsed : []);
            setRedoStack([]);
          }
        } catch {
          if (!cancelled) {
            setInkStrokes([]);
            setRedoStack([]);
          }
        }
      }

      // ANNOTATIONS
      try {
        const rows = await dbQueryAnnotations(nbId);
        const list = Array.isArray(rows) ? rows : [];
        if (!cancelled) setAnnotations(list.sort((a, b) => b.updatedAt - a.updatedAt));
      } catch {
        try {
          const raw = localStorage.getItem(LS_ANN(nbId));
          const parsed = raw ? (JSON.parse(raw) as Annotation[]) : [];
          if (!cancelled) setAnnotations(Array.isArray(parsed) ? parsed : []);
        } catch {
          if (!cancelled) setAnnotations([]);
        }
      }
    })();

    return () => {
      cancelled = true;
    };
  }, [nbId, baseDoc]);

  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  // Persist ink (debounced)
  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  useEffect(() => {
    if (!Number.isFinite(nbId) || nbId <= 0) return;
    const t = setTimeout(async () => {
      try {
        await dbPut(STORE_INK, { id: nbId, strokes: inkStrokes, updatedAt: Date.now() });
      } catch {
        localStorage.setItem(LS_INK(nbId), JSON.stringify(inkStrokes));
      }
    }, 180);
    return () => clearTimeout(t);
  }, [nbId, inkStrokes]);

  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  // Persist annotations (fallback LS) on change
  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  useEffect(() => {
    if (!Number.isFinite(nbId) || nbId <= 0) return;
    localStorage.setItem(LS_ANN(nbId), JSON.stringify(annotations));
  }, [nbId, annotations]);

  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  // Iframe: init bridge, toc, height, search, selection handlers
  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  const syncIframeHeightAndCanvas = useCallback(() => {
    const iframe = iframeRef.current;
    const canvas = canvasRef.current;
    const inner = paperInnerRef.current;
    if (!iframe || !canvas || !inner) return;

    const doc = iframe.contentDocument;
    if (!doc) return;

    // Height of iframe = document body height
    const body = doc?.body ?? null;
    const html = doc?.documentElement ?? null;
    const height = Math.max(
      body?.scrollHeight ?? 0,
      body?.offsetHeight ?? 0,
      body?.clientHeight ?? 0,
      html?.scrollHeight ?? 0,
      html?.offsetHeight ?? 0,
      html?.clientHeight ?? 0,
      1
    );
    iframe.style.height = `${height}px`;

    // Canvas should cover exactly the same area
    const cssWidth = Math.max(inner.clientWidth, 1);
    const cssHeight = Math.max(height, 1);
    const dpr = window.devicePixelRatio || 1;

    canvas.style.width = `${cssWidth}px`;
    canvas.style.height = `${cssHeight}px`;
    canvas.width = Math.floor(cssWidth * dpr);
    canvas.height = Math.floor(cssHeight * dpr);

    const ctx = canvas.getContext("2d");
    if (!ctx) return;
    ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
    ctx.clearRect(0, 0, cssWidth, cssHeight);
    inkStrokes.forEach((s) => drawStroke(ctx, s));
    if (currentStrokeRef.current) drawStroke(ctx, currentStrokeRef.current);
  }, [inkStrokes]);

  const refreshTocAndSearch = useCallback(() => {
    const doc = iframeRef.current?.contentDocument;
    if (!doc) return;

    setToc(collectTOC(doc));

    const { count, nodes } = applySearch(doc, searchQuery);
    searchNodesRef.current = nodes;
    setSearchMatchCount(count);
    setSearchMatchIndex(count > 0 ? 0 : 0);
  }, [searchQuery]);

  // Setup iframe load
  useEffect(() => {
    const iframe = iframeRef.current;
    if (!iframe) return;

    let mo: MutationObserver | null = null;
    let onClick: ((e: any) => void) | null = null;
    let onMouseUp: ((e: any) => void) | null = null;
    let onKeyUp: ((e: any) => void) | null = null;

    const setup = () => {
      const doc = iframe.contentDocument;
      if (!doc) return;

      ensureBridgeStyle(doc);

      // Build TOC
      setToc(collectTOC(doc));

      // Search highlight
      const { count, nodes } = applySearch(doc, searchQuery);
      searchNodesRef.current = nodes;
      setSearchMatchCount(count);
      setSearchMatchIndex(count > 0 ? 0 : 0);

      // Height + canvas
      syncIframeHeightAndCanvas();

      // MutationObserver to keep size correct (after highlights/pins)
      mo = new MutationObserver(() => {
        syncIframeHeightAndCanvas();
        // Keep TOC stable if headings were inserted by user content
        setToc(collectTOC(doc));
      });
      mo.observe(doc.body, { childList: true, subtree: true, characterData: true });

      // Click: pins
      onClick = (e: MouseEvent) => {
        const t = e.target as HTMLElement | null;
        if (!t) return;
        const pin = t.closest?.(".nb-pin") as HTMLElement | null;
        if (pin && pin.dataset.nbPin) {
          const pinId = pin.dataset.nbPin;
          const ann = annotations.find((a) => a.pinId === pinId);
          if (ann) {
            // set active pin
            doc.querySelectorAll(".nb-pin[data-active='true']").forEach((p) => (p as HTMLElement).dataset.active = "false");
            pin.dataset.active = "true";

            setNoteModal({
              open: true,
              mode: "edit",
              annId: ann.id,
              text: ann.text,
              quote: ann.quote,
              pinId: ann.pinId,
            });
          }
        }
      };

      // Selection popover (Amboss-like)
      const computeSelectionMenu = () => {
        if (inkMode) return; // only in study mode
        const sel = doc.getSelection();
        if (!sel || sel.rangeCount === 0) {
          selectionRangeRef.current = null;
          setSelMenu((m) => ({ ...m, open: false }));
          return;
        }
        const range = sel.getRangeAt(0);
        const text = range.toString().trim();
        if (!text || text.length < 2) {
          selectionRangeRef.current = null;
          setSelMenu((m) => ({ ...m, open: false }));
          return;
        }

        const rect = range.getBoundingClientRect();
        const iframeRect = iframe.getBoundingClientRect();

        selectionRangeRef.current = range.cloneRange();
        setSelMenu({
          open: true,
          x: iframeRect.left + rect.left + rect.width / 2,
          y: iframeRect.top + rect.top - 10,
          text: text.length > 140 ? text.slice(0, 140) + "â€¦" : text,
        });
      };

      onMouseUp = () => computeSelectionMenu();
      onKeyUp = (e: KeyboardEvent) => {
        if (e.key === "Escape") setSelMenu((m) => ({ ...m, open: false }));
        else computeSelectionMenu();
      };

      doc.addEventListener("click", onClick as any);
      doc.addEventListener("mouseup", onMouseUp as any);
      doc.addEventListener("keyup", onKeyUp as any);
    };

    // When iframe loads, setup
    const onLoad = () => setup();
    iframe.addEventListener("load", onLoad);

    // If it is already loaded
    try {
      if (iframe.contentDocument?.readyState === "complete") setup();
    } catch {
      // ignore
    }

    return () => {
      iframe.removeEventListener("load", onLoad);
      if (mo) mo.disconnect();

      const doc = iframe.contentDocument;
      if (doc) {
        if (onClick) doc.removeEventListener("click", onClick as any);
        if (onMouseUp) doc.removeEventListener("mouseup", onMouseUp as any);
        if (onKeyUp) doc.removeEventListener("keyup", onKeyUp as any);
      }
    };
  }, [annotations, inkMode, searchQuery, syncIframeHeightAndCanvas]);

  // Re-apply search when query changes
  useEffect(() => {
    const doc = iframeRef.current?.contentDocument;
    if (!doc) return;
    const { count, nodes } = applySearch(doc, searchQuery);
    searchNodesRef.current = nodes;
    setSearchMatchCount(count);
    setSearchMatchIndex(count > 0 ? 0 : 0);
    syncIframeHeightAndCanvas();
  }, [searchQuery, syncIframeHeightAndCanvas]);

  // Scroll to active search hit
  useEffect(() => {
    if (searchMatchCount <= 0) return;
    const hits = searchNodesRef.current;
    const el = hits[searchMatchIndex] as HTMLElement | undefined;
    if (!el) return;

    // compute y relative to iframe doc (outer scroll is paperViewport)
    const iframe = iframeRef.current;
    const viewport = paperViewportRef.current;
    if (!iframe || !viewport) return;

    const doc = iframe.contentDocument;
    if (!doc) return;

    // element position inside iframe document (no iframe scroll)
    const y = (el as any).offsetTop ?? 0;

    // iframe top within viewport content:
    const iframeTop = iframe.offsetTop;
    viewport.scrollTo({ top: Math.max(0, iframeTop + y - 120), behavior: "smooth" });
  }, [searchMatchIndex, searchMatchCount]);

  // Keep canvas sizing on resize
  useEffect(() => {
    const onResize = () => syncIframeHeightAndCanvas();
    window.addEventListener("resize", onResize);
    return () => window.removeEventListener("resize", onResize);
  }, [syncIframeHeightAndCanvas]);

  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  // Canvas redraw helper
  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  const redrawCanvas = useCallback(
    (preview?: InkStroke | null) => {
      const canvas = canvasRef.current;
      const inner = paperInnerRef.current;
      if (!canvas || !inner) return;
      const ctx = canvas.getContext("2d");
      if (!ctx) return;

      const dpr = window.devicePixelRatio || 1;
      ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
      ctx.clearRect(0, 0, inner.clientWidth, canvas.height / dpr);

      inkStrokes.forEach((stroke) => drawStroke(ctx, stroke));
      if (preview) drawStroke(ctx, preview);
    },
    [inkStrokes]
  );

  useEffect(() => {
    redrawCanvas(currentStrokeRef.current);
  }, [inkStrokes, redrawCanvas]);

  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  // Canvas pointer (ink mode)
  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  const getEventPoint = useCallback((e: React.PointerEvent<HTMLCanvasElement>): InkPoint => {
    const rect = e.currentTarget.getBoundingClientRect();
    return { x: e.clientX - rect.left, y: e.clientY - rect.top };
  }, []);

  const onPointerDown = useCallback(
    (e: React.PointerEvent<HTMLCanvasElement>) => {
      if (!inkMode) return;
      const p = getEventPoint(e);
      isDrawingRef.current = true;
      e.currentTarget.setPointerCapture(e.pointerId);

      const stroke: InkStroke = {
        tool: inkTool,
        color: inkTool === "highlighter" ? "#f59e0b" : "#111827",
        size: inkTool === "highlighter" ? 10 : inkTool === "eraser" ? 18 : 2.6,
        points: [p],
      };

      currentStrokeRef.current = stroke;
    },
    [getEventPoint, inkMode, inkTool]
  );

  const onPointerMove = useCallback(
    (e: React.PointerEvent<HTMLCanvasElement>) => {
      if (!inkMode || !isDrawingRef.current || !currentStrokeRef.current) return;
      const p = getEventPoint(e);
      currentStrokeRef.current.points.push(p);
      redrawCanvas(currentStrokeRef.current);
    },
    [getEventPoint, inkMode, redrawCanvas]
  );

  const finishStroke = useCallback(() => {
    if (!isDrawingRef.current) return;
    isDrawingRef.current = false;

    const stroke = currentStrokeRef.current;
    currentStrokeRef.current = null;

    if (!stroke || stroke.points.length < 2) {
      redrawCanvas(null);
      return;
    }

    setInkStrokes((prev) => [...prev, stroke]);
    setRedoStack([]);
  }, [redrawCanvas]);

  const undoInk = useCallback(() => {
    setInkStrokes((prev) => {
      if (prev.length === 0) return prev;
      const next = prev.slice(0, -1);
      const popped = prev[prev.length - 1];
      setRedoStack((r) => [...r, popped]);
      return next;
    });
  }, []);

  const redoInk = useCallback(() => {
    setRedoStack((prev) => {
      if (prev.length === 0) return prev;
      const restored = prev[prev.length - 1];
      setInkStrokes((strokes) => [...strokes, restored]);
      return prev.slice(0, -1);
    });
  }, []);

  const clearInk = useCallback(() => {
    setInkStrokes([]);
    setRedoStack([]);
